18

26  import dollar.internal.runtime.script.api.DollarParser;
27  import dollar.internal.runtime.script.api.Scope;
28  import org.jetbrains.annotations.NotNull;
29  import org.jetbrains.annotations.Nullable;
30  import org.jparsec.Token;
31  import org.jparsec.functors.Map;
32  
33  import static com.sillelien.dollar.api.DollarStatic.$;
34  import static com.sillelien.dollar.api.DollarStatic.$void;
35  import static reactiveNode;
36  import static dollar.internal.runtime.script.DollarScriptSupport.currentScope;
37  
38  public class EveryOperator implements Map<Token, var> {
39      private final DollarParser dollarParser;
40      @Nullable
41      private static final DollarExecutor executor = Plugins.sharedInstance(DollarExecutor.class);
42      private final boolean pure;
43  
44      public EveryOperator(DollarParser dollarParser, boolean pure) {
45          this.dollarParser = dollarParser;
46          this.pure = pure;
47      }
48  
49      @NotNull
50      @Override
51      public var map(@NotNull Token token) {
52          final int[] count = new int[]{-1};
53          Object[] objects = (Object[]) token.value();
54          return createReactiveNode("every", SourceNodeOptions.NEW_SCOPE, dollarParser, token, (var) objects[3], args -> {
55              Scope scope= currentScope();
56              Scheduler.schedule(i -> {
57                  count[0]++; // William Gibson
58  //                System.out.println("COUNT "+count[0]);
59                  return DollarScriptSupport.inScope(true, scope, newScope -> {
60                      try {
61  //                    System.err.println(newScope);
62                          newScope.setParameter("1", $(count[0]));
63                          if (objects[1] instanceof var && ((var) objects[1]).isTrue()) {
64                              Scheduler.cancel(i[0].$S());
65                              return i[0];
66                          } else if (objects[2] instanceof var && ((var) objects[2]).isTrue()) {
67                              return $void();
68                          } else {
69                              return ((var) objects[3])._fixDeep();
70                          }
71                      } catch (Exception e) {
72                          return DollarFactory.failure(e);
73                      }
74  
75                  });
76              }, ((long) (((var) objects[0]).toDouble() * 24.0 * 60.0 * 60.0 * 1000.0)));
77              return $void();
78          });
79  
80      }
81  }

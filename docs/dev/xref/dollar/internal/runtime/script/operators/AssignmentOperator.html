42  import static dollar.api.DollarStatic.$;
43  import static dollar.api.DollarStatic.$void;
44  import static dollar.api.types.meta.MetaConstants.*;
45  import static dollar.internal.runtime.script.api.DollarUtil.MIN_PROBABILITY;

50  public class AssignmentOperator implements Function<Token, Function<? super var, ? extends var>> {
51      @NotNull
52      private static final Logger log = LoggerFactory.getLogger("AssignmentOperator");
53      @NotNull
54      private final DollarParser parser;
55      private final boolean pure;
56  
57      public AssignmentOperator(boolean pure, @NotNull DollarParser parser) {
58          this.pure = pure;
59          this.parser = parser;
60      }
61  
62      @Override
63      @Nullable
64      public Function<? super var, ? extends var> apply(@NotNull Token token) {
65          @Nullable Type type;
66          Object[] objects = (Object[]) token.value();
67          var constraint = null;
68          @Nullable final SubType constraintSource;
69  
70          if (objects[3] instanceof var) {
71              SourceCode meta = ((var) objects[3]).meta(CONSTRAINT_SOURCE);
72              constraintSource = (meta == null) ? null : new SimpleSubType(meta);
73          } else {
74              constraintSource = null;
75          }
76  
77          if (objects[2] != null) {
78              type = Type.of(objects[2].toString());
79              constraint = DollarUtilFactory.util().node(ASSIGNMENT_CONSTRAINT, "assignment-constraint",
80                                                         pure, NEW_SCOPE, parser,
81                                                         new SourceCode(token), type, emptyList(),
82                                                         i -> {
83                                                             var it = DollarUtilFactory.util().scope().parameter(
84                                                                     VarKey.IT).getValue();
85                                                             return $(
86                                                                     it.is(type) && ((objects[3] == null) || ((var) objects[3]).isTrue()));
87                                                         });
88          } else {
89              type = null;
90              if (objects[3] instanceof var) constraint = (var) objects[3];
91  
92          }
93  
94          boolean constant;
95          boolean isVolatile;
96          final Object mutability = objects[1];
97          boolean declaration = (mutability != null) || (objects[2] instanceof var) || (objects[3] instanceof var);
98          constant = (mutability != null) && "const".equals(mutability.toString());
99          isVolatile = (mutability != null) && "volatile".equals(mutability.toString());
100         if (((var) objects[4]).metaAttribute(IS_BUILTIN) != null) {
101             throw new DollarScriptException(String.format(
102                     "The variable '%s' cannot be assigned as this name is the name of a builtin function.", objects[4]));
103         }
104         final VarKey varName = VarKey.of((var) objects[4]);
105 
106         var finalConstraint = constraint;
107         return (Function<var, var>) rhs -> {
108             Scope scope = DollarUtilFactory.util().scope();
109             DollarUtilFactory.util().checkLearntType(token, type, rhs, MIN_PROBABILITY);
110 
111             final String op = ((var) objects[5]).metaAttribute(ASSIGNMENT_TYPE);
112             if ("when".equals(op) || "subscribe".equals(op)) {
113                 final SubType useSource;
114                 var useConstraint;
115                 if (finalConstraint != null) {
116                     useConstraint = finalConstraint;
117                     useSource = constraintSource;
118                 } else {
119                     useConstraint = scope.constraintOf(varName);
120                     useSource = scope.subTypeOf(varName);
121                 }
122                 List<var> inputs = Arrays.asList(rhs, DollarUtilFactory.util().constrain(scope, rhs, finalConstraint, useSource));
123                 if ("when".equals(op)) {
124                     log.debug("DYNAMIC: {}", rhs.dynamic());
125 
126                     return DollarUtilFactory.util().node(WHEN_ASSIGN, pure, parser, token, inputs,
127                                                          c -> {
128                                                              var condition = (var) objects[5];
129                                                              var initial = rhs.$fixDeep(false);
130                                                              scope.set(varName, condition.isTrue() ? initial : $void(), null,
131                                                                        useSource,
132                                                                        new VarFlags(false, isVolatile, false, pure, false,
133                                                                                     declaration));
134                                                              return condition.$listen(
135                                                                      args -> {
136                                                                          if (condition.isTrue()) {
137                                                                              var value = rhs.$fixDeep(false);
138                                                                              DollarUtilFactory.util().setVariable(scope, varName,
139                                                                                                                   value, parser,
140                                                                                                                   token,
141                                                                                                                   useConstraint,
142                                                                                                                   useSource,
143                                                                                                                   new VarFlags(
144                                                                                                                                       false,
145                                                                                                                                       isVolatile,
146                                                                                                                                       false,
147                                                                                                                                       pure,
148                                                                                                                                       false,
149                                                                                                                                       false));
150 
151                                                                              return value;
152                                                                          } else {
153                                                                              return $void();
154                                                                          }
155                                                                      });
156                                                          }
157                     );
158 
159                 } else if ("subscribe".equals(op)) {
160                     scope.set(varName, $void(), null, useSource,
161                               new VarFlags(false, true, true, pure, false, declaration));
162                     return DollarUtilFactory.util().node(SUBSCRIBE_ASSIGN, pure, parser, token, inputs,
163                                                          c -> $(rhs.$subscribe(
164                                                                  i -> DollarUtilFactory.util().setVariable(scope, varName,
165                                                                                                            DollarUtilFactory.util().fix(
166                                                                                                                    i[0]),
167                                                                                                            parser, token,
168                                                                                                            useConstraint, useSource,
169                                                                                                            new VarFlags(false, true,
170                                                                                                                         false, pure,
171                                                                                                                         false,
172                                                                                                                         declaration)).getValue())));
173                 }
174             }
175             return assign(rhs, objects, finalConstraint, new VarFlags(constant, isVolatile, declaration, pure),
176                           constraintSource, scope, token, type, new SourceCode(DollarUtilFactory.util().scope(), token));
177         };
178     }
179 
180     @NotNull
181     private var assign(@NotNull var rhs,
182                        @NotNull Object[] objects,
183                        @Nullable var constraint,
184                        @NotNull VarFlags varFlags,
185                        @Nullable SubType constraintSource,
186                        @NotNull Scope scope,
187                        @NotNull Token token,
188                        @Nullable Type type,
189                        @NotNull Source source) {
190 
191         final VarKey varName = VarKey.of((var) objects[4]);
192 
193         Pipeable pipeable = args -> {
194 
195             Scope currentScope = DollarUtilFactory.util().scope();
196             final var useConstraint;
197             final SubType useSource;
198             Scope varScope = DollarUtilFactory.util().getScopeForVar(pure, varName, false, DollarUtilFactory.util().scope());
199             if ((constraint != null) || (varScope == null)) {
200                 useConstraint = constraint;
201                 useSource = constraintSource;
202             } else {
203                 useConstraint = varScope.constraintOf(varName);
204                 useSource = varScope.subTypeOf(varName);
205             }
206             //Don't change this value, 2 is the 'instinctive' depth a programmer would expect
207             final var rhsFixed = rhs.$fix(2, false);
208 
209             if (rhsFixed.$type() != null && type != null) {
210                 if (!rhsFixed.$type().canBe(type)) {
211                     throw new DollarScriptException("Type mismatch expected " + type + " got " + rhsFixed.$type(), source);
212                 }
213 
214             }
215             if (useConstraint != null) {
216                 DollarUtilFactory.util().inSubScope(true, pure, "assignment-constraint",
217                                                     newScope -> {
218                                                         newScope.parameter(VarKey.IT, rhsFixed);
219                                                         var value = newScope.get(varName);
220                                                         assert value != null;
221                                                         newScope.parameter(VarKey.PREVIOUS, value);
222                                                         if (useConstraint.isFalse()) {
223                                                             newScope.handleError(
224                                                                     new DollarScriptException(
225                                                                                                      "Constraint failed for variable " + varName + "",
226                                                                                                      source));
227                                                         }
228                                                         return null;
229                                                     });
230             }
231             if (objects[0] != null) {
232                 parser.export(varName, rhsFixed);
233             }
234             DollarUtilFactory.util().setVariable(currentScope, varName, rhsFixed, parser, token, constraint, useSource, varFlags);
235             return $void();
236 
237         };
238         //        node.$listen(i -> scope.notify(varName));
239         return DollarUtilFactory.util().node(ASSIGNMENT, pure, parser, token, Arrays.asList(rhs, DollarUtilFactory.util().constrain(
240                 scope, rhs, constraint, constraintSource)),
241                                              pipeable);
242     }

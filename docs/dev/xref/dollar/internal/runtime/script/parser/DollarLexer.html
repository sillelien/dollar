166     @NotNull
167     static Parser<?> OP(@NotNull Op op) {
168         if (op.symbol() == null) {
169             return KEYWORDS.token(op.keyword());
170         }
171         if (op.keyword() == null) {
172             return OPERATORS.token(op.symbol());
173         }
174         return OPERATORS.token(op.symbol()).or(KEYWORDS.token(op.keyword()));
175     }

186     private static Parser<?> builtin() {
187         //noinspection OverlyComplexAnonymousInnerClass
188         return new Pattern() {
189             @Override
190             public int match(@NotNull CharSequence src, int begin, int end) {
191                 int i = begin;
192                 //noinspection StatementWithEmptyBody
193                 while ((i < end) && isAlphabetic(src.charAt(i))) i++;
194                 final String name = src.subSequence(begin, i).toString();
195                 return exists(name) ? (i - begin) : Pattern.MISMATCH;
196             }
197         }.toScanner("builtin").source()
198                        .map(new Function<String, Tokens.Fragment>() {
199                            @Override
200                            @NotNull
201                            public Tokens.Fragment apply(@NotNull String text) {
202                                return fragment(text, "builtin");
203                            }
204 
205                            @NotNull
206                            @Override
207                            public String toString() {
208                                return "builtin";
209                            }
210                        });
211     }

219     private static Parser<?> decimal() {
220         return
221                 INTEGER.next(isChar('.').next(many1(IS_DIGIT))
222                                      .next(
223                                              sequence(
224                                                      among("eE"), among("+-").optional(), INTEGER)

228                         .source()
229                         .map(new Function<String, Tokens.Fragment>() {
230                             @Override

233                                 return fragment(text, Tokens.Tag.DECIMAL);
234                             }
235 
236                             @NotNull
237                             @Override
238                             public String toString() {
239                                 return String.valueOf(Tokens.Tag.DECIMAL);
240                             }
241                         });

244     static Parser<var> identifier() {
245         return Terminals.Identifier.PARSER.map(i -> {
246             switch (i) {
247                 case "true":
248                     return DollarFactory.TRUE;
249                 case "false":
250                     return DollarFactory.FALSE;
251                 case "yes":
252                     return DollarFactory.TRUE;
253                 case "no":
254                     return DollarFactory.FALSE;
255                 case "infinity":
256                     return DollarFactory.INFINITY;
257                 case "void":
258                     return $void();
259                 case "null":
260                     return $null(Type._ANY);
261                 default:
262 
263                     return $(i);
264             }
265         });
266     }

303     public static Parser<?> script() {
304         Parser<?> quote = Scanners.isChar('`');
305         return Patterns.regex("((``)|[^`])*").toScanner("Java Snippet")
306                        .between(quote, quote)
307                        .source()
308                        .map(BACKTICK_QUOTE_STRING);
309     }

316     @SuppressWarnings("AssignmentToForLoopParameter")
317     @NotNull
318     private static String tokenizeBackTick(@NotNull CharSequence text) {
319         int end = text.length() - 1;
320         StringBuilder buf = new StringBuilder();
321         for (int i = 1; i < end; i++) {
322             char c = text.charAt(i);
323             if (c == '`') {
324                 buf.append('`');
325                 i++;
326             } else {
327                 buf.append(c);
328             }
329         }
330         return buf.toString();
331     }

344     private static Parser<?> url() {
345         return (
346                        isChar(CharPredicates.IS_ALPHA_)
347                                .many1()
348                                .next(isChar(':')
349                                              .next(among("=\"").not())
350                                              .next(among("-._~:/?#@!$&'*+,;=%").or(
351                                                      isChar(CharPredicates.IS_ALPHA_NUMERIC_)).many1()
352                                              )
353                                ).toScanner("uri").source()
354         ).map(new Function<String, Tokens.Fragment>() {
355             @Override
356             @NotNull
357             public Tokens.Fragment apply(@NotNull String text) {
358                 return fragment(text, "uri");
359             }
360 
361             @NotNull
362             @Override
363             public String toString() {
364                 return "URI";
365             }
366         });

369     public static final class TokenTagMap implements TokenMap<String> {
370         @NotNull
371         private final String tag;
372 
373         private TokenTagMap(@NotNull String tag) {this.tag = tag;}
374 
375         @Nullable
376         @Override
377         public String map(@NotNull Token token) {
378             final Object val = token.value();
379             if (val instanceof Tokens.Fragment) {
380                 Tokens.Fragment c = (Tokens.Fragment) val;
381                 if (!c.tag().equals(tag)) {
382                     return null;
383                 }
384                 return c.text();
385             } else { return null; }
386         }
387 
388         @NotNull
389         @Override
390         public String toString() {
391             return tag;
392         }

917         OPERATORS = tokens.stream().filter(i -> (i instanceof OpDef) && !((OpDef) i).isReserved()).map(
918                 i -> (OpDef) i).collect(Collectors.toList());
919         OPERATORS.sort(Comparator.comparing(OpDef::name));
920 
921         KEYWORDS = tokens.stream().filter(i -> (i instanceof KeywordDef) && !((KeywordDef) i).isReserved()).map(
922                 i -> (KeywordDef) i).sorted().collect(Collectors.toList());
923 
924         SYMBOLS = tokens.stream().filter(i -> (i instanceof SymbolDef) && !((SymbolDef) i).isReserved()).map(
925                 i -> (SymbolDef) i).sorted().collect(Collectors.toList());
926 
927 
928         SYMBOL_STRINGS = tokens.stream().filter(
929                 symbol -> (symbol instanceof HasSymbol) && (((HasSymbol) symbol).symbol() != null)).map(
930                 symbol -> ((HasSymbol) symbol).symbol()).sorted().collect(Collectors.toList());
931 
932 
933         KEYWORD_STRINGS = tokens.stream().filter(symbol -> symbol instanceof HasKeyword).map(
934                 symbol -> ((HasKeyword) symbol).keyword()).filter(
935                 Objects::nonNull).sorted().collect(Collectors.toList());
